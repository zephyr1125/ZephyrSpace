# 商业航天公司市值矩形树图

面积 = 市值　｜　颜色 = 细分赛道　｜　数据来自各公司页 frontmatter

> [!warning] 数据说明
> **Amazon** 条目为母公司整体市值（含 AWS、电商等全部业务），并非 Kuiper 项目单独估值。
> **SpaceX、蓝箭航天、银河航天** 为非上市参考估值，基于公开融资信息推算，仅供研究参考，不构成投资建议。

> [!tip] 依赖插件
> 此视图需要社区插件 **obsidian-echarts**（作者：cumany）已安装并启用。

```dataviewjs
// ── 非上市公司参考估值（单位：亿美元）────────────────────────
const PRIVATE_REF = {
  "SpaceX":   { value: 3500, sector: "多赛道综合",    note: "参考估值（未上市，基于融资轮次推算，约 3500 亿美元）" },
  "蓝箭航天": { value: 5,    sector: "商业发射",       note: "参考估值（一级市场，约 35 亿元人民币）" },
  "银河航天": { value: 15,   sector: "低轨卫星互联网", note: "参考估值（一级市场，约 100 亿元人民币）" },
};

const DISPLAY_NAME = {
  "Amazon Project Kuiper": "Amazon",
};

const SECTOR_COLORS = {
  "低轨卫星互联网": "#3a7fd5",
  "多赛道综合":     "#7b4fb5",
  "综合航天":       "#27a869",
  "卫星通信":       "#c0392b",
  "商业发射":       "#d47c1a",
};

function parseMV(str) {
  if (!str) return null;
  const s = String(str);
  const wan = s.match(/([\d.]+)\s*万亿/);
  const yi  = s.match(/([\d.]+)\s*亿/);
  if (wan) return parseFloat(wan[1]) * 10000;
  if (yi)  return parseFloat(yi[1]);
  return null;
}

function fmtMV(v) {
  if (v >= 10000) return `${(v / 10000).toFixed(2)} 万亿美元`;
  return `${v} 亿美元`;
}

// ── 从公司页 frontmatter 组装数据 ────────────────────────────
const treeData = [];

for (const p of dv.pages('"01-公司"')) {
  const fileName = p.file.name;
  const dispName = DISPLAY_NAME[fileName] || fileName;

  if (PRIVATE_REF[fileName]) {
    const ref = PRIVATE_REF[fileName];
    treeData.push({
      name:      dispName,
      value:     ref.value,
      itemStyle: { color: SECTOR_COLORS[ref.sector] || "#555" },
      tooltip:   { formatter: () =>
        `<b>${dispName}</b><br/>细分赛道：${ref.sector}<br/>` +
        `市值：<b>${fmtMV(ref.value)}</b><br/>股票：未上市<br/>` +
        `可投资性：${p.可投资性 || "一级市场"}<br/>` +
        `<span style="color:#9ca3af;font-size:11px">注：${ref.note}</span>`
      },
    });
    continue;
  }

  const mv = parseMV(p.市值 || p.母公司市值);
  if (!mv) continue;

  const isParentStock = !!p.母公司市值;
  const sectors  = Array.isArray(p.细分赛道) ? p.细分赛道 : [p.细分赛道].filter(Boolean);
  const sector   = sectors[0] || "其他";
  const ticker   = [p.股票代码 || p.母公司股票代码, p.交易所].filter(Boolean).join(" · ");
  const pe       = p.市盈率 ?? p.母公司市盈率 ?? null;
  const peStr    = pe != null ? `<br/>市盈率（TTM）：<b>${pe}</b>` : "";
  const noteStr  = isParentStock ? `<br/><span style="color:#9ca3af;font-size:11px">注：此处为母公司 Amazon 完整市值，Kuiper 为旗下子项目</span>` : "";
  const updStr   = p.数据更新日期 ? `<br/><span style="color:#6b7280;font-size:11px">数据截至 ${p.数据更新日期}</span>` : "";

  treeData.push({
    name:      dispName,
    value:     mv,
    itemStyle: { color: SECTOR_COLORS[sector] || "#555" },
    tooltip:   { formatter: () =>
      `<b>${dispName}</b><br/>细分赛道：${sector}<br/>` +
      `市值：<b>${fmtMV(mv)}</b>（${mv} 亿美元）<br/>` +
      `股票：${ticker || "-"}<br/>可投资性：${p.可投资性 || "-"}` +
      peStr + noteStr + updStr
    },
  });
}

treeData.sort((a, b) => b.value - a.value);

// ── 通过 obsidian-echarts 插件实例渲染 ────────────────────────
const plugin = app.plugins.plugins["obsidian-echarts"];

if (!plugin) {
  dv.el("p", "⚠️ 未找到 obsidian-echarts 插件，请确认已安装并启用。", {
    attr: { style: "color:#f87171;" }
  });
} else {
  // 用容器实际宽度，保持原始宽高比（900:520 ≈ 1.73:1）
  const w = this.container.clientWidth || 700;
  const h = Math.round(w / 1.73);

  plugin.render({
    width:  w,
    height: h,
    backgroundColor: "transparent",
    tooltip: { confine: true },
    series: [{
      type:        "treemap",
      roam:        false,
      nodeClick:   false,
      width:       "100%",
      height:      "100%",
      breadcrumb:  { show: false },
      visibleMin:  0,
      label: {
        show:       true,
        fontSize:   13,
        fontWeight: "bold",
        color:      "rgba(255,255,255,0.9)",
        formatter:  p => `${p.name}\n${fmtMV(p.data.value)}`,
      },
      upperLabel:  { show: false },
      itemStyle:   { gapWidth: 3, borderWidth: 0 },
      data: treeData,
    }],
  }, this.container);
}
```

## 数据维护说明

- 上市公司市值和市盈率从各自的[[01-公司]]页面 frontmatter 读取（字段：`市值` / `母公司市值` / `市盈率`），**更新公司页即可自动反映到树图**。
- SpaceX、银河航天、蓝箭航天的参考估值硬编码在本页脚本中，需要手动更新。
- 字段格式示例：`市值: 386亿美元`、`母公司市值: 2.24万亿美元`。
